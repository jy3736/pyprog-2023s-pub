def cntUnique(s):
    values = [int(x) for x in s.split()]
    count = 0
    for i in range(len(values) - 1):
        if values[i] != values[i + 1]:
            count += 1
    count += 1
    return count

def swapMinMax(n):
    min_i, max_i = 0, 0
    
    for i in range(len(n)):
        if n[i] < n[min_i]:
            min_i = i
        if n[i] > n[max_i]:
            max_i = i
    n[min_i], n[max_i] = n[max_i], n[min_i]
    return n

arr = [int(x) for x in input().split()]
arr = [str(x) for x in swapMinMax(arr)]
result = ' '.join(arr)
print(result)

def snowflake(n):
    arr = [["." for _ in range(n)] for _ in range(n)]
    mid = n // 2
    for i in range(n):
        arr[mid][i] = "*"
        arr[i][mid] = "*"
        arr[i][i] = "*"
        arr[i][n-1-i] = "*"
    return arr

def print_arr(arr):
    for row in arr:
        print(" ".join(row))
        

print_arr(snowflake(int(input())))

def read_input():
    m, n, r = map(int, input().split())
    A = [list(map(int, input().split())) for _ in range(m)]
    B = [list(map(int, input().split())) for _ in range(n)]
    return A, B

def matrix_product(A, B):
    m, n = len(A), len(B)
    r = len(B[0])
    AB = [[0] * r for _ in range(m)]
    for i in range(m):
        for j in range(r):
            dot_product = 0
            for k in range(n):
                dot_product += A[i][k] * B[k][j]
            AB[i][j] = dot_product
    return AB

def print_matrix(matrix):
    for row in matrix:
        print(*row)
