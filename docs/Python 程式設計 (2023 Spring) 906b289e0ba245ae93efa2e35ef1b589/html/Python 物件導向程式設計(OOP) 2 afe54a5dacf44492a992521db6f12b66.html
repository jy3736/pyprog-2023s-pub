Defining and Using Classes

# Define a class called Person
class Person:
    # Define the constructor method with two parameters
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    # Define a method to print the person's name and age
    def print_info(self):
        print(f"My name is {self.name} and I am {self.age} years old.")

# Create two instances of the Person class
person1 = Person("John", 30)
person2 = Person("Jane", 25)

# Call the print_info method on both instances
person1.print_info()
person2.print_info()

Output of the Code

My name is John and I am 30 years old.
My name is Jane and I am 25 years old.

„ÄêË™≤Â†ÇÁ∑¥Áøí 3„ÄëSimple Inheritance

Defining and Using Classes

# Define a parent class called Animal
class Animal:
    # Define the constructor method with one parameter
    def __init__(self, name):
        self.name = name
    
    # Define a method to print the animal's name
    def print_name(self):
        print(f"My name is {self.name}.")

# Define a child class called Dog that inherits from Animal
class Dog(Animal):
    # Define the constructor method with two parameters
    def __init__(self, name, breed):
        super().__init__(name)
        self.breed = breed
    
    # Define a method to print the dog's breed
    def print_breed(self):
        print(f"I am a {self.breed}.")

# Create an instance of the Dog class
my_dog = Dog("Fido", "Labrador")

# Call the print_name and print_breed methods on the instance
my_dog.print_name()
my_dog.print_breed()

Output of the Code

My name is Fido.
I am a Labrador.
