import math

def count_primes(A, B):
    count = 0
    for num in range(A, B+1):
        if num > 1:
            for i in range(2, int(math.sqrt(num))+1):
                if num % i == 0:
                    break
            else:
                count += 1
    return count

a = int(input())
b = int(input())

print(count_primes(a,b))
