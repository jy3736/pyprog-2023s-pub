def reverse_h(s):
    first_h = s.index("h")
    last_h = s.rindex("h")
    if first_h != last_h:
        s = s[:first_h + 1] + s[first_h + 1:last_h][::-1] + s[last_h:]
    return s

print(reverse_h(input()))

def max_idx(lst):
    m = max(lst)
    if lst.count(m) == 1:
        return lst.index(m)+1
    for i in range(len(lst)):
        if lst[i] == m:
            return i+1

def max_clen(sequence):
    max_length = 0
    curr_length = 0
    curr_num = None
    for num in sequence:
        if num == curr_num:
            curr_length += 1
        else:
            curr_num = num
            curr_length = 1
        max_length = max(max_length, curr_length)
    return max_length
